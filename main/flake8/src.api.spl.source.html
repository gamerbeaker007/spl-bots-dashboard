104 def compute_sig(string_to_sign: str, private_key: str):
105 bytestring_signature = sign_message(string_to_sign, private_key)
106 sig = hexlify(bytestring_signature).decode('ascii')
107 return sig

110 def get_token(username: str, private_key: str):
111 login_endpoint = base_url + 'players/v2/login'
112 ts = int(time() * 1000)
113 sig = compute_sig(username + str(ts), private_key)
114 params = {
115 'name': username,
116 'ts': ts,
117 'sig': sig
118 }
119 result = http.get(login_endpoint, params=params)
120 token = ""
121 version = ""
122 if result and result.status_code == 200:
123 result = result.json()
124 if 'error' in result:
125 raise ValueError(result['error'])
126  
127 version = result['timestamp']
128 token = result['token']
129 return token, version
